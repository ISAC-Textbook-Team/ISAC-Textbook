\section*{Exercises}
\addcontentsline{toc}{section}{Exercises}

\subsection*{Section 2.1}

\begin{enumerate}[label=\textbf{\arabic*.}, leftmargin=*]
    \item \textbf{Link Budget, SNR, and Capacity.}

    Consider a wireless communication link with carrier frequency
    \(f_c=3\,\mathrm{GHz}\), propagation distance \(d=1\,\mathrm{km}\),
    transmit power \(P_t=0\,\mathrm{dBm}\), system bandwidth
    \(B=10\,\mathrm{MHz}\), and noise power spectral density
    \(N_0=-174\,\mathrm{dBm/Hz}\). Assume free-space propagation and unit
    antenna gains.
    Compute the free-space path loss, received power, noise power, SNR, and
    Shannon capacity.

    \item \textbf{SINR and Throughput.}

    A receiver observes useful signal power \(P_s=-80\,\mathrm{dBm}\),
    interference power \(P_i=-86\,\mathrm{dBm}\), and noise power
    \(P_n=-90\,\mathrm{dBm}\). If the nominal transmission rate is
    \(R=20\,\mathrm{Mbit/s}\) and the block error rate is
    \(\mathrm{BLER}=0.1\), compute the SINR and the approximate actual
    throughput.

    \item \textbf{Coherence Time and Coherence Bandwidth.}

    Consider a mobile link with carrier frequency \(f_c=28\,\mathrm{GHz}\),
    relative velocity \(v=30\,\mathrm{m/s}\), symbol duration
    \(T_s=20\,\mu\mathrm{s}\), root-mean-square delay spread
    \(\tau_{\mathrm{rms}}=0.5\,\mu\mathrm{s}\), and signal bandwidth
    \(B=100\,\mathrm{MHz}\). Using the approximate relationships
    \(T_c\approx 1/f_D\) and \(B_c\approx 1/\tau_{\mathrm{rms}}\), determine
    whether the channel is closer to slow or fast fading, and whether it is
    closer to flat or frequency-selective fading.

    \item \textbf{MIMO Signal Model and Spatial Streams.}

    A MIMO-OFDM system has \(N_t=4\) transmit antennas and \(N_r=2\) receive
    antennas. On one subcarrier, the channel matrix has
    \(\operatorname{rank}(\mathbf{H}[k])=2\), and the transmitter sends
    \(N_s=2\) spatial streams.
    Give the dimensions of \(\mathbf{H}[k]\), \(\mathbf{W}[k]\),
    \(\mathbf{s}[k]\), \(\mathbf{x}[k]\), and \(\mathbf{y}[k]\). How many
    spatial streams can be supported at most on this subcarrier?

\end{enumerate}
